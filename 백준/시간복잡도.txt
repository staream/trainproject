# 일차원 그래프의 기울기에 의한 대소비교 생각하기

a1,a2= map(int,input().split())
c= int(input())
n0=int(input())
F= a1*n0 + a2
G= c*n0
if a1<=c:
    if F<=G:
        print("1")
    else:
        print("0")
else:
    print("0")
